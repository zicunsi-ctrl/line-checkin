<div class="card">
    <h2>歡迎蒞臨活動現場</h2>
    <p>系統正在讀取您的 LINE 資訊...</p>
    
    <div id="info-box" style="display:none;">
        <p class="user-info">您好，<strong id="username">載入中...</strong></p>
        <p style="font-size:14px; color:#666;">請點擊下方按鈕完成無紙化報到</p>
        <button id="btn-checkin" onclick="submitCheckIn()">確認報到</button>
    </div>
</div>

<script>
    // 填入您在 LINE Developer 申請的 LIFF ID
    const MY_LIFF_ID = '您的LIFF_ID'; 
    // 填入剛剛 Google Apps Script 部署後的 URL
    const GAS_WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbzu_o-Q8xReDcWAGoNa5gyE2BsjEsQQ-MfU4ZcgYgN0_ofhjLKNoqxDTrZpHshyWiA-/exec'; 

    let userProfile = {};

    // 初始化 LINE LIFF
    async function initLiff() {
        try {
            await liff.init({ liffId: MY_LIFF_ID });
            if (liff.isLoggedIn()) {
                // 取得使用者 LINE 檔案
                userProfile = await liff.getProfile();
                document.getElementById('username').innerText = userProfile.displayName;
                document.getElementById('info-box').style.display = 'block';
            } else {
                liff.login(); // 引導登入
            }
        } catch (error) {
            alert('LIFF 初始化失敗: ' + error);
        }
    }

    // 送出報到資料到後端 Excel
    async function submitCheckIn() {
        const btn = document.getElementById('btn-checkin');
        btn.disabled = true;
        btn.innerText = '報到傳送中...';

        const payload = {
            userId: userProfile.userId,
            displayName: userProfile.displayName
        };

        try {
            const response = await fetch(GAS_WEB_APP_URL, {
                method: 'POST',
                body: JSON.stringify(payload)
            });
            const result = await response.json();
            
            if(result.status === 'success') {
                alert('報到成功！感謝您的參與。');
                liff.closeWindow(); // 自動關閉 LINE 網頁視窗
            } else {
                alert('報到失敗，請洽現場工作人員。');
                btn.disabled = false;
                btn.innerText = '確認報到';
            }
        } catch (error) {
            alert('連線錯誤: ' + error);
            btn.disabled = false;
            btn.innerText = '確認報到';
        }
    }

    // 執行初始化
    initLiff();
</script>
